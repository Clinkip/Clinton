import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.metrics import roc_curve, auc

def draw_roc_curve(model, test_X, test_y):
    # Ensure test_X is a NumPy array
    test_X = np.array(test_X)

    # Predict probabilities
    if hasattr(model, 'predict_proba'):
        predict_proba = model.predict_proba(test_X)
        
        # Check if predict_proba has at least two columns
        if predict_proba.shape[1] > 1:
            predict_proba_df = pd.DataFrame(predict_proba)
            test_results_df = pd.DataFrame({'actual': test_y}).reset_index(drop=True)
            test_results_df['status'] = predict_proba_df.iloc[:, 1]
        else:
            # Handle binary classification case
            test_results_df = pd.DataFrame({'actual': test_y, 'status': predict_proba[:, 0]})
    else:
        # Fallback to decision function for models that don't have predict_proba (like KNeighborsClassifier)
        predict_scores = model.decision_function(test_X)
        test_results_df = pd.DataFrame({'actual': test_y, 'status': predict_scores})

    # Compute ROC curve
    fpr, tpr, _ = roc_curve(test_results_df['actual'], test_results_df['status'])
    roc_auc = auc(fpr, tpr)

    # Plot ROC curve
    plt.figure()
    plt.plot(fpr, tpr, color='blue', lw=2, label='ROC curve (area = %0.2f)' % roc_auc)
    plt.plot([0, 1], [0, 1], color='navy', lw=2, linestyle='--')
    plt.xlim([0.0, 1.0])
    plt.ylim([0.0, 1.05])
    plt.xlabel('False Positive Rate')
    plt.ylabel('True Positive Rate')
    plt.title('Receiver Operating Characteristic')
    plt.legend(loc="lower right")
    plt.show()

# Example usage
# Assuming knn_clf is your KNeighborsClassifier model, and test_X and test_y are your test datasets
# Ensure knn_clf is fitted/trained before calling this function
draw_roc_curve(knn_clf, test_X, test_y)

import numpy as np
from sklearn.neighbors import KNeighborsClassifier
from sklearn.metrics import confusion_matrix
import seaborn as sn
import matplotlib.pyplot as plt

# Assuming knn_clf is your KNeighborsClassifier model and test_X, test_y are your test datasets

def draw_cm(actual, predicted):
    cm = confusion_matrix(actual, predicted, labels=[1, 0])
    sn.heatmap(cm, annot=True, fmt='.2f',
               xticklabels=["Placed", "Not Placed"],
               yticklabels=["Placed", "Not Placed"])
    plt.ylabel('True label')
    plt.xlabel('Predicted label')
    plt.show()

# Convert test_X to a NumPy array
test_X = np.array(test_X)

# Predict using the KNN classifier
pred_y = knn_clf.predict(test_X)

# Draw the confusion matrix
draw_cm(test_y, pred_y)
